Quick start

Post with a few
lines of code

One API call. Every platform. Node.js, Python, Go, PHP, cURL — pick your language.

ayrshare-quickstart 
// npm install @ayrshare/social-post-api

const SocialPost = require('@ayrshare/social-post-api');
const api = new SocialPost('YOUR_API_KEY');

const response = await api.post({
  post: "Today is a great day!",
  platforms: ["twitter", "facebook", "instagram", "linkedin"],
  scheduleDate: new Date().toISOString(),
});

console.log(response.id);       // "post_abc123"
console.log(response.postIds);  // { twitter: "...", facebook: "..." }

from ayrshare import SocialPost

api = SocialPost("YOUR_API_KEY")

response = api.post({
    "post": "Today is a great day!",
    "platforms": ["twitter", "facebook", "instagram"],
    "scheduleDate": "2026-02-19T12:00:00Z",
})

print(response["id"])       # post_abc123
print(response["postIds"])  # {"twitter": "...", "facebook": "..."}

// go get github.com/ayrshare/ayrshare-go

package main

import (
  "fmt"
  ayrshare "github.com/ayrshare/ayrshare-go"
)

func main() {
  client := ayrshare.NewClient("YOUR_API_KEY")
  resp, _ := client.Post(ayrshare.PostRequest{
    Post:      "Today is a great day!",
    Platforms: []string{"twitter", "facebook"},
  })
  fmt.Println(resp.ID) // post_abc123
}
